Type definition¶

type node =
  { mutable name : string
  ; mutable prev : node array
  ; mutable next : node array
  ; mutable neuron : neuron
  ; mutable output : t option
  ; mutable network : network
  ; mutable train : bool
  }

  and network =
  { mutable nnid : string
  ; mutable size : int
  ; mutable roots : node array
  ; mutable outputs : node array
  ; mutable topo : node array
  }
